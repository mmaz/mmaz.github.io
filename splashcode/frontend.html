data Discrete = Down | Up deriving (Eq)

discreteButton :: MonadWidget t m => String -> m (Event t Discrete)
discreteButton btntext = do
  let attrs = ("class" =: "waves-effect waves-light pink accent-2 btn") <> ("style" =: "font-size:30px")
  (btnEl, _) <- elAttr' "button" attrs $ text btntext
  return $ leftmost [const Down <$> domEvent Mousedown btnEl, const Up <$> domEvent Mouseup btnEl, const Up <$> domEvent Mouseleave btnEl] -- NB: Mouseout is *not* what you want

data BotAPI = BotAPI String String
forward, backward, cw, ccw, stop :: BotAPI
forward = BotAPI "↑" "forward"
backward = BotAPI "↓" "backward"
cw = BotAPI "⟳" "cw"
ccw = BotAPI "⟲" "ccw"
stop = BotAPI "⇏" stopurl
lower = BotAPI "▼" "lower"
higher = BotAPI "▲" "higher"
stopurl :: String
stopurl = "stop"

controlParrot :: MonadWidget t m => BotAPI -> m ()
controlParrot (BotAPI icon url) = do
  evDis <- discreteButton icon
  let start = xhrRequest "GET" url def
  let end = xhrRequest "GET" stopurl def
  performRequestAsync $ const start <$> ffilter (== Down) evDis
  performRequestAsync $ const end <$> ffilter (== Up) evDis
  blank

quad :: MonadWidget t m => m ()
quad = do
  elAttr "div" img $ elAttr "div" ("style" =: "font-size: 50px; width: 612px; height: 500px;") $ do
    divClass "container" $ do
      divClass "row" $ do
        divClass "col s4 center-align" $ controlParrot lower
        divClass "col s4 center-align" $ controlParrot stop
        divClass "col s4 center-align" $ controlParrot higher
      divClass "row" $ divClass "col s12 center-align" $ controlParrot forward
      divClass "row" $ do
        divClass "col s6 center-align" $ controlParrot ccw
        divClass "col s6 center-align" $ controlParrot cw
      divClass "row" $ divClass "col s12 center-align" $ controlParrot backward
      divClass "row" $ divClass "col s12 center-align" $ divClass "chip" $ text "Haskell drone controller"
  where img = "style" =: "width: 612px; margin: 0 auto; background-image: url(\"rs.jpg\"); background-repeat: no-repeat; background-size: 612px; "

connect :: MonadWidget t m => m ()
connect = divClass "row" $ do
  let a1 = "class" =: "waves-effect waves-light green btn"
      a2 = "class" =: "waves-effect waves-light orange btn"
      a3 = "class" =: "waves-effect waves-light red btn"
      a4 = "class" =: "waves-effect waves-light purple btn"
  (b1, _) <- elAttr' "button" a1 $ text "Takeoff"
  (b2, _) <- elAttr' "button" a2 $ text "Land"
  (b3, _) <- elAttr' "button" a3 $ text "Emergency"
  (b4, _) <- elAttr' "button" a4 $ text "Quit"
  performRequestAsync $ pure (xhrRequest "GET" "takeoff" def) <$> (domEvent Click b1)
  performRequestAsync $ pure (xhrRequest "GET" "land" def) <$> (domEvent Click b2)
  performRequestAsync $ pure (xhrRequest "GET" "emergency" def) <$> (domEvent Click b3)
  performRequestAsync $ pure (xhrRequest "GET" "quit" def) <$> (domEvent Click b4)
  blank

main :: IO ()
main = mainWidget $ divClass "container" $ do
  divClass "row" $ blank
  quad
  elAttr "div" ("style" =: "width: 480px; margin: 0 auto;") $ connect
  blank
